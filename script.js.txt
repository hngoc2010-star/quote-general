const inputs=document.querySelectorAll("input,select")

inputs.forEach(el=>{
el.addEventListener("input",handleInput)
})

function parseNum(v){
return Number(v.replace(/,/g,""))||0
}

function formatNum(v){
return v.toLocaleString("en-US",{maximumFractionDigits:2})
}

function formatInput(el){

let start=el.selectionStart
let value=el.value.replace(/,/g,"")

if(value==="") return

let num=Number(value)

if(isNaN(num)) return

let formatted=num.toLocaleString("en-US")

el.value=formatted

let newPos=formatted.length
el.setSelectionRange(newPos,newPos)

}

function handleInput(e){

let el=e.target

if(el.type!=="number"){
formatInput(el)
}

updateSize()

calculate()

}

function updateSize(){

let size=document.getElementById("size").value

if(size==="1220x2440"){
width.value=1220
length.value=2440
}

if(size==="915x1830"){
width.value=915
length.value=1830
}

}

function calculate(){

let thickness=parseNum(thickness.value)

let widthM=parseNum(width.value)/1000
let lengthM=parseNum(length.value)/1000

let bundles=parseNum(bundles.value)
let pcsBundle=parseNum(pcsBundle.value)

let pricePc=parseNum(pricePc.value)
let priceCbm=parseNum(priceCbm.value)

let usd=parseNum(usdRate.value)
let logisticsCost=parseNum(logistics.value)

let cfrPrice=parseNum(cfr.value)

let rule=parseNum(rule.value)

let thicknessSell=thickness+rule

let totalPcs=bundles*pcsBundle

let cbmBuy=widthM*lengthM*(thickness/1000)*totalPcs
let cbmSell=widthM*lengthM*(thicknessSell/1000)*totalPcs

let cbmDiff=cbmSell-cbmBuy

let exw=0

if(pricePc>0){
exw=pricePc*totalPcs
}else{
exw=priceCbm*cbmBuy
}

let revenue=cfrPrice*usd*cbmSell

let totalCost=exw+logisticsCost

let profit=revenue-totalCost

let profitCbm=cbmSell?profit/cbmSell:0

totalPcs.innerText=formatNum(totalPcs)
cbmBuy.innerText=formatNum(cbmBuy)
cbmSell.innerText=formatNum(cbmSell)
cbmDiff.innerText=formatNum(cbmDiff)

profitTotal.innerText=formatNum(profit)
profitCbm.innerText=formatNum(profitCbm)

if(profit<0){
profitTotal.className="red"
profitCbm.className="red"
}else{
profitTotal.className="blue"
profitCbm.className="blue"
}

}
