document.addEventListener("DOMContentLoaded", () => {

const inputs = document.querySelectorAll("input,select")

inputs.forEach(el => {
el.addEventListener("input", calculate)
})

function num(v){
return Number(v)||0
}

function format(v){
return v.toLocaleString("en-US",{maximumFractionDigits:2})
}

function calculate(){

let thicknessBuy=num(document.getElementById("thicknessBuy").value)
let bundles=num(document.getElementById("bundles").value)
let pcsBundle=num(document.getElementById("pcsBundle").value)
let pricePc=num(document.getElementById("pricePc").value)
let priceCbm=num(document.getElementById("priceCbm").value)
let usdRate=num(document.getElementById("usdRate").value)
let logistics=num(document.getElementById("logistics").value)
let cfr=num(document.getElementById("cfrPrice").value)

let size=document.getElementById("size").value
let rule=document.getElementById("rule").value

let width=0
let length=0

if(size=="1220x2440"){
width=1.22
length=2.44
}else{
width=0.915
length=2.405
}

let adjust=0

if(rule=="plus") adjust=0.5
if(rule=="equal") adjust=0
if(rule=="minus") adjust=-0.5

let thicknessSell=thicknessBuy+adjust

let totalPcs=bundles*pcsBundle

let cbmBuy=width*length*(thicknessBuy/1000)*totalPcs
let cbmSell=width*length*(thicknessSell/1000)*totalPcs

let cbmDiff=cbmSell-cbmBuy

let exw=0

if(pricePc>0){
exw=pricePc*totalPcs
}else{
exw=priceCbm*cbmBuy
}

let revenue=cfr*usdRate*cbmSell

let totalCost=exw+logistics

let profit=revenue-totalCost

let profitCbm=cbmSell?profit/cbmSell:0

document.getElementById("totalPcs").innerText=format(totalPcs)
document.getElementById("cbmBuy").innerText=format(cbmBuy)
document.getElementById("cbmSell").innerText=format(cbmSell)
document.getElementById("cbmDiff").innerText=format(cbmDiff)

let profitCbmEl=document.getElementById("profitCbm")
let profitTotalEl=document.getElementById("profitTotal")

profitCbmEl.innerText=format(profitCbm)
profitTotalEl.innerText=format(profit)

if(profit<0){
profitTotalEl.className="red"
profitCbmEl.className="red"
}else{
profitTotalEl.className="blue"
profitCbmEl.className="blue"
}

}

})
