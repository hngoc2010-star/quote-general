const inputs=document.querySelectorAll("input,select")

inputs.forEach(el=>{
el.addEventListener("input",calculate)
})

document.getElementById("size").addEventListener("change",updateSize)

function num(v){
return Number(v)||0
}

function format(v){
return v.toLocaleString("en-US",{maximumFractionDigits:2})
}

function updateSize(){

let size=document.getElementById("size").value

let widthBox=document.getElementById("customWidthBox")
let lengthBox=document.getElementById("customLengthBox")

let width=document.getElementById("width")
let length=document.getElementById("length")

if(size==="1220x2440"){

widthBox.style.display="none"
lengthBox.style.display="none"

width.value=1220
length.value=2440

}

else if(size==="915x1830"){

widthBox.style.display="none"
lengthBox.style.display="none"

width.value=915
length.value=1830

}

else{

widthBox.style.display="block"
lengthBox.style.display="block"

}

calculate()

}

function calculate(){

let thickness=num(document.getElementById("thickness").value)

let width=num(document.getElementById("width").value)/1000
let length=num(document.getElementById("length").value)/1000

let bundles=num(document.getElementById("bundles").value)
let pcsBundle=num(document.getElementById("pcsBundle").value)

let pricePc=num(document.getElementById("pricePc").value)
let priceCbm=num(document.getElementById("priceCbm").value)

let usd=num(document.getElementById("usdRate").value)
let logistics=num(document.getElementById("logistics").value)

let cfr=num(document.getElementById("cfr").value)

let rule=num(document.getElementById("rule").value)

let thicknessSell=thickness+rule

let totalPcs=bundles*pcsBundle

let cbmBuy=width*length*(thickness/1000)*totalPcs
let cbmSell=width*length*(thicknessSell/1000)*totalPcs

let cbmDiff=cbmSell-cbmBuy

let exw=0

if(pricePc>0){
exw=pricePc*totalPcs
}else{
exw=priceCbm*cbmBuy
}

let revenue=cfr*usd*cbmSell

let totalCost=exw+logistics

let profit=revenue-totalCost

let profitCbm=cbmSell?profit/cbmSell:0

document.getElementById("totalPcs").innerText=format(totalPcs)
document.getElementById("cbmBuy").innerText=format(cbmBuy)
document.getElementById("cbmSell").innerText=format(cbmSell)
document.getElementById("cbmDiff").innerText=format(cbmDiff)

let profitTotal=document.getElementById("profitTotal")
let profitCbmEl=document.getElementById("profitCbm")

profitTotal.innerText=format(profit)
profitCbmEl.innerText=format(profitCbm)

if(profit<0){
profitTotal.className="red"
profitCbmEl.className="red"
}else{
profitTotal.className="blue"
profitCbmEl.className="blue"
}

}

updateSize()
